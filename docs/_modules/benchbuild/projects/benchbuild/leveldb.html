[docs]    def build(self):
        clang = cc(self)
        clang_cxx = cxx(self)
        with local.cwd(self.SRC_FILE):
            with local.env(CXX=str(clang_cxx), CC=str(clang)):
                make("clean")
                run(make["all", "-i"])
