[docs]    def build(self):
        """
        Build process for OpenMP enabled LULESH code:

        Required: openmp (omp.h) needs to be available.
        """
        self.cflags += ["-fopenmp"]

        clang_cxx = lt_clang_cxx(self.cflags, self.ldflags,
                                 self.compiler_extension)
        run(clang_cxx["-o", self.run_f, self.SRC_FILE])
