[docs]    def compile(self):
        emerge_in_chroot = uchroot.uchroot()["/usr/bin/emerge"]
        prog = self.DOMAIN + "/" + str(self.NAME)[len(self.DOMAIN) + 1:]
        with local.env(CONFIG_PROTECT="-*"):
            emerge_in_chroot(
                "--autounmask-only=y",
                "--autounmask-write=y",
                prog,
                retcode=None)
        uchroot.uretry(emerge_in_chroot[prog])
