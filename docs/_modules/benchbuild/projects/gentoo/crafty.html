[docs]    def download(self):
        super(Crafty, self).download()

        book_file = "book.bin"
        book_bin = "http://www.craftychess.com/" + book_file
        Wget(book_bin, book_file)



[docs]    def build(self):
        emerge_in_chroot = uchroot()["/usr/bin/emerge"]
        run(emerge_in_chroot["games-board/crafty"])
