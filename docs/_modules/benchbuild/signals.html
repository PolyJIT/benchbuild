handlers = CleanupOnSignal()

def __handle_sigterm(signum, frame):
    LOG.debug("Got SIGTERM, running cleanup handlers")
    handlers()
    sys.exit(signum)

signal.signal(signal.SIGTERM, __handle_sigterm)
