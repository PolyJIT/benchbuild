[docs]def set_defaults():
    """Configure the loggers default settings."""
    log_levels = {
        3: logging.DEBUG,
        2: logging.INFO,
        1: logging.WARNING,
        0: logging.ERROR
    }

    logging.captureWarnings(True)
    LOG = logging.getLogger()
    LOG.setLevel(log_levels[settings.CFG["verbosity"].value()])
