[docs]def parse_schedule_tree(tree_str):
    if tree_str is None:
        return "No Input"

    res = None
    try:
        ret = _ROOT.parseString(tree_str)
        return str(ret[0])
    except p.ParseException as ex:
        LOG.warning("Failed to parse:")
        LOG.warning(str(ex))
        LOG.warning(tree_str)
        return None
